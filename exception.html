Exception Handling

Q1: Handle NO_DATA_FOUND exception

DECLARE v_name EMP.ENAME%TYPE; BEGIN SELECT ENAME INTO v_name FROM EMP WHERE EMPNO=9999; DBMS_OUTPUT.PUT_LINE(v_name); EXCEPTION WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('No employee found'); END;

Q2: Handle TOO_MANY_ROWS exception

DECLARE v_name EMP.ENAME%TYPE; BEGIN SELECT ENAME INTO v_name FROM EMP WHERE JOB='CLERK'; EXCEPTION WHEN TOO_MANY_ROWS THEN DBMS_OUTPUT.PUT_LINE('Too many rows returned'); END;

Q3: Handle ZERO_DIVIDE exception

DECLARE v_result NUMBER; BEGIN v_result:=10/0; EXCEPTION WHEN ZERO_DIVIDE THEN DBMS_OUTPUT.PUT_LINE('Division by zero not allowed'); END;

Q4: Handle VALUE_ERROR exception

DECLARE v_num NUMBER; BEGIN v_num:='ABC'; EXCEPTION WHEN VALUE_ERROR THEN DBMS_OUTPUT.PUT_LINE('Invalid value error'); END;

Q5: Generic WHEN OTHERS handler

BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error occurred'); END;

Q6: Display SQLCODE and SQLERRM

BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error Code='||SQLCODE); DBMS_OUTPUT.PUT_LINE('Error Message='||SQLERRM); END;

Q7: Capture error details for division by zero

DECLARE v_result NUMBER; BEGIN v_result:=10/0; EXCEPTION WHEN ZERO_DIVIDE THEN DBMS_OUTPUT.PUT_LINE('Code='||SQLCODE||' Msg='||SQLERRM); END;

Q8: Log exception message into variable

DECLARE v_msg VARCHAR2(200); BEGIN v_msg:=''; v_msg:=TO_CHAR(10/0); EXCEPTION WHEN OTHERS THEN v_msg:=SQLERRM; DBMS_OUTPUT.PUT_LINE(v_msg); END;

Q9: Raise error if salary < 2000

DECLARE v_sal NUMBER:=1500; BEGIN IF v_sal<2000 THEN RAISE_APPLICATION_ERROR(-20009,'Salary too low'); END IF; END;

Q10: Validate age and raise error if < 18

DECLARE v_age NUMBER:=16; BEGIN IF v_age<18 THEN RAISE_APPLICATION_ERROR(-20010,'Age must be 18+'); END IF; END;

Q11: Restrict update operation

BEGIN UPDATE EMP SET JOB='CLERK'; RAISE_APPLICATION_ERROR(-20011,'Update not allowed'); END;

Q12: Raise error for invalid department

DECLARE v_dept NUMBER:=50; BEGIN IF NOT EXISTS (SELECT 1 FROM DEPT WHERE DEPTNO=v_dept) THEN RAISE_APPLICATION_ERROR(-20012,'Invalid Department'); END IF; END;

Q13: Declare and handle user-defined exception

DECLARE e_invalid EXCEPTION; v_sal NUMBER:=1000; BEGIN IF v_sal<2000 THEN RAISE e_invalid; END IF; EXCEPTION WHEN e_invalid THEN DBMS_OUTPUT.PUT_LINE('Custom Exception Raised'); END;

Q14: Raise custom exception when salary exceeds limit

DECLARE e_high_sal EXCEPTION; v_sal NUMBER:=10000; BEGIN IF v_sal>8000 THEN RAISE e_high_sal; END IF; EXCEPTION WHEN e_high_sal THEN DBMS_OUTPUT.PUT_LINE('Salary exceeds limit'); END;

Q15: Validate input and raise user-defined exception

DECLARE e_invalid_data EXCEPTION; v_age NUMBER:=0; BEGIN IF v_age<=0 THEN RAISE e_invalid_data; END IF; EXCEPTION WHEN e_invalid_data THEN DBMS_OUTPUT.PUT_LINE('Invalid Age Entered'); END;

Q16: Handle multiple user-defined exceptions

DECLARE e_low_sal EXCEPTION; e_high_sal EXCEPTION; v_sal NUMBER:=9000; BEGIN IF v_sal<2000 THEN RAISE e_low_sal; ELSIF v_sal>8000 THEN RAISE e_high_sal; END IF; EXCEPTION WHEN e_low_sal THEN DBMS_OUTPUT.PUT_LINE('Salary too low'); WHEN e_high_sal THEN DBMS_OUTPUT.PUT_LINE('Salary too high'); END;

Q17: Catch exception and re-raise

BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error caught, re-raising...'); RAISE; END;

Q18: Inner block handles, re-raises to outer block

BEGIN BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Inner block handled'); RAISE; END; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Outer block handled'); END;

Q19: Log error and re-raise

BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN INSERT INTO ERROR_LOG VALUES(SQLCODE,SQLERRM,SYSDATE); RAISE; END;

Q20: Fetch employee details and handle all exceptions

DECLARE v_name EMP.ENAME%TYPE; BEGIN SELECT ENAME INTO v_name FROM EMP WHERE EMPNO=9999; DBMS_OUTPUT.PUT_LINE(v_name); EXCEPTION WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('No employee found'); WHEN TOO_MANY_ROWS THEN DBMS_OUTPUT.PUT_LINE('Multiple employees found'); WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error: '||SQLERRM); END;

Q21: User-defined exception + RAISE_APPLICATION_ERROR

Q24: Simulate bank withdrawal with exception handling

DECLARE v_balance NUMBER:=5000; v_withdraw NUMBER:=6000; BEGIN IF v_withdraw>v_balance THEN RAISE_APPLICATION_ERROR(-20014,'Insufficient Balance'); ELSE v_balance:=v_balance-v_withdraw; DBMS_OUTPUT.PUT_LINE('Remaining Balance='||v_balance); END IF; END;

Q25: Validate employee data and raise different exceptions (Multiple Exception Handling)

DECLARE e_invalid_age EXCEPTION; e_invalid_sal EXCEPTION; v_age NUMBER:=15; v_sal NUMBER:=1000; BEGIN IF v_age<18 THEN RAISE e_invalid_age; ELSIF v_sal<2000 THEN RAISE e_invalid_sal; END IF; EXCEPTION WHEN e_invalid_age THEN DBMS_OUTPUT.PUT_LINE('Age must be 18+'); WHEN e_invalid_sal THEN DBMS_OUTPUT.PUT_LINE('Salary must be at least 2000'); END;

Q26: Log errors into a table using exception handling (Logging Errors)

BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN INSERT INTO ERROR_LOG VALUES(SQLCODE,SQLERRM,SYSDATE); DBMS_OUTPUT.PUT_LINE('Error logged'); END;

Q27: Nested exception blocks with re-raise mechanism (Advanced Exception Handling)

BEGIN BEGIN UPDATE EMP SET SAL=SAL/0; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Inner block handled'); RAISE; END; EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Outer block handled and re-raised'); END;
