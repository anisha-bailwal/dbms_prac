% Type, % Rowtype

Q1: Fetch employee name from EMP table for empno=7369 (SELECT INTO)

DECLARE v_name EMP.ENAME%TYPE; BEGIN SELECT ENAME INTO v_name FROM EMP WHERE EMPNO=7369; DBMS_OUTPUT.PUT_LINE('Employee Name: '||v_name); END;

Q2: Retrieve salary of an employee and print it (SELECT INTO)

DECLARE v_sal EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=7499; DBMS_OUTPUT.PUT_LINE('Salary: '||v_sal); END;

Q3: Fetch employee name and job using a single SELECT INTO

DECLARE v_name EMP.ENAME%TYPE; v_job EMP.JOB%TYPE; BEGIN SELECT ENAME, JOB INTO v_name, v_job FROM EMP WHERE EMPNO=7521; DBMS_OUTPUT.PUT_LINE(v_name||' - '||v_job); END;

Q4: Display department number of an employee (SELECT INTO)

DECLARE v_dept EMP.DEPTNO%TYPE; BEGIN SELECT DEPTNO INTO v_dept FROM EMP WHERE EMPNO=7566; DBMS_OUTPUT.PUT_LINE('Dept No: '||v_dept); END;

Q5: Fetch hiredate of an employee and display it (SELECT INTO)

DECLARE v_date EMP.HIREDATE%TYPE; BEGIN SELECT HIREDATE INTO v_date FROM EMP WHERE EMPNO=7839; DBMS_OUTPUT.PUT_LINE('Hire Date: '||v_date); END;

Q6: Count total number of employees (SELECT INTO with aggregate)

DECLARE v_count NUMBER; BEGIN SELECT COUNT(*) INTO v_count FROM EMP; DBMS_OUTPUT.PUT_LINE('Total Employees: '||v_count); END;

Q10: Calculate annual salary using %TYPE variable

DECLARE v_sal EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=7839; DBMS_OUTPUT.PUT_LINE('Annual Salary: '||(v_sal*12)); END;

Q11: Update salary using %TYPE variable

DECLARE v_sal EMP.SAL%TYPE := 5000; BEGIN UPDATE EMP SET SAL=v_sal WHERE EMPNO=7902; DBMS_OUTPUT.PUT_LINE('Salary Updated'); END;

Q13: Declare a record using %ROWTYPE for EMP table

DECLARE v_emp EMP%ROWTYPE; BEGIN SELECT * INTO v_emp FROM EMP WHERE EMPNO=7566; DBMS_OUTPUT.PUT_LINE(v_emp.ENAME||' '||v_emp.SAL||' '||v_emp.JOB); END;

Q16: Update salary using %ROWTYPE variable

DECLARE v_emp EMP%ROWTYPE; BEGIN SELECT * INTO v_emp FROM EMP WHERE EMPNO=7654; v_emp.SAL := v_emp.SAL+500; UPDATE EMP SET SAL=v_emp.SAL WHERE EMPNO=v_emp.EMPNO; DBMS_OUTPUT.PUT_LINE('Salary Updated: '||v_emp.SAL); END;

Q17: Insert a record using %ROWTYPE

DECLARE v_emp EMP%ROWTYPE; BEGIN v_emp.EMPNO := 9001; v_emp.ENAME := 'ANISHA'; v_emp.JOB := 'CLERK'; v_emp.SAL := 2500; v_emp.DEPTNO := 20; INSERT INTO EMP VALUES v_emp; DBMS_OUTPUT.PUT_LINE('Record Inserted'); END;
