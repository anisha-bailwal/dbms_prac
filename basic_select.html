Basic SELECT Queries

Q1: Display all columns of the EMP table

SELECT * FROM EMP;

Q2: Display EMPNO, ENAME, and SAL of all employees from the EMP table

SELECT EMPNO, ENAME, SAL FROM EMP;

Q3: Display unique job titles from the EMP table

SELECT DISTINCT JOB FROM EMP;

Q4: Display employee names along with their department numbers from the EMP table

SELECT ENAME, DEPTNO FROM EMP;

Q5: Display all department details from the DEPT table

SELECT * FROM DEPT;

Q6: Display distinct department numbers from the EMP table

SELECT DISTINCT DEPTNO FROM EMP;

Q7: Display employee name and annual salary (SAL × 12) from the EMP table

SELECT ENAME, SAL*12 AS ANNUAL_SALARY FROM EMP;

Q8: Display employee name, salary, and commission from the EMP table

SELECT ENAME, SAL, COMM FROM EMP;

Q9: Display job and salary of all employees from the EMP table

SELECT JOB, SAL FROM EMP;

Q10: Display employees along with their manager numbers from the EMP table

SELECT ENAME, MGR FROM EMP;
