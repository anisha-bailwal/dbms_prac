Group Functions

Q1: Display the total number of employees

SELECT COUNT(*) AS Total_Employees FROM EMP;

Q2: Display the total salary of all employees

SELECT SUM(SAL) AS Total_Salary FROM EMP;

Q3: Display the average salary of all employees

SELECT AVG(SAL) AS Average_Salary FROM EMP;

Q4: Display the highest salary

SELECT MAX(SAL) AS Highest_Salary FROM EMP;

Q5: Display the lowest salary

SELECT MIN(SAL) AS Lowest_Salary FROM EMP;

Q6: Display the total number of departments in EMP table

SELECT COUNT(DISTINCT DEPTNO) AS Dept_Count FROM EMP;

Q7: Display the number of employees who are managers

SELECT COUNT(*) AS Managers FROM EMP WHERE JOB = 'MANAGER';

Q8: Display the total commission paid to employees

SELECT SUM(COMM) AS Total_Commission FROM EMP;

Q9: Display the average commission

SELECT AVG(COMM) AS Average_Commission FROM EMP;

Q10: Display the number of employees who are getting commission

SELECT COUNT(*) AS Commissioned_Employees FROM EMP WHERE COMM IS NOT NULL;

Q11: Display department-wise total salary

SELECT DEPTNO, SUM(SAL) AS Total_Salary FROM EMP GROUP BY DEPTNO;

Q12: Display department-wise average salary

SELECT DEPTNO, AVG(SAL) AS Avg_Salary FROM EMP GROUP BY DEPTNO;

Q13: Display department-wise number of employees

Q26: Display department and job having highest total salary

SELECT DEPTNO, JOB, SUM(SAL) AS Total_Salary FROM EMP GROUP BY DEPTNO, JOB ORDER BY Total_Salary DESC FETCH FIRST 1 ROWS ONLY;

Q27: Display department where sum of salaries is maximum

SELECT DEPTNO, SUM(SAL) AS Total_Salary FROM EMP GROUP BY DEPTNO ORDER BY Total_Salary DESC FETCH FIRST 1 ROWS ONLY;

Q28: Display job having minimum average salary

SELECT JOB, AVG(SAL) AS Avg_Salary FROM EMP GROUP BY JOB ORDER BY Avg_Salary ASC FETCH FIRST 1 ROWS ONLY;

Q29: Display jobs having average salary greater than department-wise average salary

SELECT JOB, AVG(SAL) FROM EMP GROUP BY JOB HAVING AVG(SAL) > (SELECT AVG(SAL) FROM EMP GROUP BY DEPTNO);

Q30: Display department having highest number of employees

SELECT DEPTNO, COUNT(*) AS Emp_Count FROM EMP GROUP BY DEPTNO ORDER BY Emp_Count DESC FETCH FIRST 1 ROWS ONLY;

Q31: Display jobs whose total salary is more than salary of all CLERKs combined

SELECT JOB, SUM(SAL) FROM EMP GROUP BY JOB HAVING SUM(SAL) > (SELECT SUM(SAL) FROM EMP WHERE JOB='CLERK');

Q32: Display departments having same number of employees

SELECT DEPTNO, COUNT(*) AS Emp_Count FROM EMP GROUP BY DEPTNO HAVING COUNT(*) IN (SELECT COUNT(*) FROM EMP GROUP BY DEPTNO HAVING COUNT(*) > 1);

Q33: Display jobs where max salary is equal to department 20 max salary

SELECT JOB, MAX(SAL) FROM EMP GROUP BY JOB HAVING MAX(SAL) = (SELECT MAX(SAL) FROM EMP WHERE DEPTNO=20);
