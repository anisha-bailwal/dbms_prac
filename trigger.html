Triggers

Q1: Trigger to display a message whenever a record is inserted into EMP (Trigger Basics)

CREATE OR REPLACE TRIGGER trg_insert_msg AFTER INSERT ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Record Inserted into EMP'); END;

Q2: BEFORE INSERT trigger on EMP

CREATE OR REPLACE TRIGGER trg_before_insert BEFORE INSERT ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Before Insert Trigger Fired'); END;

Q3: AFTER DELETE trigger on EMP

CREATE OR REPLACE TRIGGER trg_after_delete AFTER DELETE ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Record Deleted from EMP'); END;

Q4: Prevent deletion from EMP (BEFORE DELETE)

CREATE OR REPLACE TRIGGER trg_prevent_delete BEFORE DELETE ON EMP BEGIN RAISE_APPLICATION_ERROR(-20001,'Deletion not allowed'); END;

Q5: AFTER UPDATE trigger to display message

CREATE OR REPLACE TRIGGER trg_after_update AFTER UPDATE ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Employee Salary Updated'); END;

Q6: Row-level trigger to display old and new salary

CREATE OR REPLACE TRIGGER trg_salary_change AFTER UPDATE OF SAL ON EMP FOR EACH ROW BEGIN DBMS_OUTPUT.PUT_LINE('Old Salary='||:OLD.SAL||' New Salary='||:NEW.SAL); END;

Q7: Restrict salary update if new salary < old salary

CREATE OR REPLACE TRIGGER trg_validate_salary BEFORE UPDATE OF SAL ON EMP FOR EACH ROW BEGIN IF :NEW.SAL<:OLD.SAL THEN RAISE_APPLICATION_ERROR(-20002,'New salary cannot be less than old salary'); END IF; END;

Q8: Automatically set hiredate while inserting employee record

CREATE OR REPLACE TRIGGER trg_set_hiredate BEFORE INSERT ON EMP FOR EACH ROW BEGIN :NEW.HIREDATE:=SYSDATE; END;

Q9: Increase salary by 5% before updating

CREATE OR REPLACE TRIGGER trg_increase_salary BEFORE UPDATE OF SAL ON EMP FOR EACH ROW BEGIN :NEW.SAL:=:OLD.SAL*1.05; END;

Q10: Copy deleted records into audit table

CREATE OR REPLACE TRIGGER trg_audit_delete AFTER DELETE ON EMP FOR EACH ROW BEGIN INSERT INTO EMP_AUDIT VALUES(:OLD.EMPNO,:OLD.ENAME,SYSDATE,'DELETE'); END;

Q11: Statement-level trigger to display message on insert

CREATE OR REPLACE TRIGGER trg_stmt_insert AFTER INSERT ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Insert Statement Executed'); END;

Q12: Statement-level trigger to log update operations

CREATE OR REPLACE TRIGGER trg_stmt_update AFTER UPDATE ON EMP BEGIN INSERT INTO EMP_LOG VALUES(USER,SYSDATE,'UPDATE'); END;

Q13: Trigger fires once for DELETE statement

CREATE OR REPLACE TRIGGER trg_stmt_delete BEFORE DELETE ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Delete Statement Fired'); END;

Q14: BEFORE trigger to validate salary before inserting

CREATE OR REPLACE TRIGGER trg_validate_insert BEFORE INSERT ON EMP FOR EACH ROW BEGIN IF :NEW.SAL<0 THEN RAISE_APPLICATION_ERROR(-20003,'Salary must be positive'); END IF; END;

Q15: AFTER trigger to log inserted employee records

CREATE OR REPLACE TRIGGER trg_log_insert AFTER INSERT ON EMP FOR EACH ROW BEGIN INSERT INTO EMP_LOG VALUES(:NEW.EMPNO,:NEW.ENAME,SYSDATE,'INSERT'); END;

Q16: BEFORE and AFTER triggers for UPDATE

CREATE OR REPLACE TRIGGER trg_before_update BEFORE UPDATE ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('Before Update Trigger'); END; / CREATE OR REPLACE TRIGGER trg_after_update2 AFTER UPDATE ON EMP BEGIN DBMS_OUTPUT.PUT_LINE('After Update Trigger'); END;

Q17: Trigger fires only when salary > 5000

CREATE OR REPLACE TRIGGER trg_conditional AFTER INSERT OR UPDATE OF SAL ON EMP FOR EACH ROW WHEN (NEW.SAL>5000) BEGIN DBMS_OUTPUT.PUT_LINE('High Salary Employee'); END;

Q18: Trigger to allow insert only during working hours

CREATE OR REPLACE TRIGGER trg_working_hours BEFORE INSERT ON EMP FOR EACH ROW BEGIN IF TO_CHAR(SYSDATE,'HH24') NOT BETWEEN 9 AND 17 THEN RAISE_APPLICATION_ERROR(-20004,'Insert allowed only during working hours'); END IF; END;

Q19: Trigger to prevent update on weekends

CREATE OR REPLACE TRIGGER trg_no_weekend_update BEFORE UPDATE ON EMP BEGIN IF TO_CHAR(SYSDATE,'DY') IN ('SAT','SUN') THEN RAISE_APPLICATION_ERROR(-20005,'Update not allowed on weekends'); END IF; END;

Q20: Trigger to maintain audit table for INSERT, UPDATE, DELETE

CREATE OR REPLACE TRIGGER trg_audit_all AFTER INSERT OR UPDATE OR DELETE ON EMP FOR EACH ROW BEGIN IF INSERTING THEN INSERT INTO EMP_AUDIT VALUES(:NEW.EMPNO,:NEW.ENAME,SYSDATE,'INSERT'); ELSIF UPDATING THEN INSERT INTO EMP_AUDIT VALUES(:OLD.EMPNO,:OLD.ENAME,SYSDATE,'UPDATE'); ELSIF DELETING THEN INSERT INTO EMP_AUDIT VALUES(:OLD.EMPNO,:OLD.ENAME,SYSDATE,'DELETE'); END IF; END;

Q21: Trigger to generate automatic employee ID

CREATE OR REPLACE TRIGGER trg_auto_empid BEFORE INSERT ON EMP FOR EACH ROW BEGIN SELECT emp_seq.NEXTVAL INTO :NEW.EMPNO FROM dual; END;

Q22: Trigger to enforce referential integrity manually

CREATE OR REPLACE TRIGGER trg_integrity BEFORE INSERT ON EMP FOR EACH ROW BEGIN IF NOT EXISTS (SELECT 1 FROM DEPT WHERE DEPTNO=:NEW.DEPTNO) THEN RAISE_APPLICATION_ERROR(-20006,'Invalid Department Number'); END IF; END;

Q23: Trigger to calculate bonus after salary update

CREATE OR REPLACE TRIGGER trg_bonus_calc AFTER UPDATE OF SAL ON EMP FOR EACH ROW BEGIN UPDATE EMP SET COMM=:NEW.SAL*0.1 WHERE EMPNO=:NEW.EMPNO; END;

Q23: Trigger to calculate bonus after salary update

CREATE OR REPLACE TRIGGER trg_calc_bonus AFTER UPDATE OF SAL ON EMP FOR EACH ROW BEGIN UPDATE EMP SET COMM=:NEW.SAL*0.1 WHERE EMPNO=:NEW.EMPNO; END;

Q24: Trigger to track number of rows affected (Advanced Logic)

CREATE OR REPLACE TRIGGER trg_row_count AFTER UPDATE ON EMP DECLARE v_count NUMBER; BEGIN SELECT SQL%ROWCOUNT INTO v_count FROM dual; DBMS_OUTPUT.PUT_LINE('Rows affected='||v_count); END;

Q25: Trigger to prevent duplicate employee names (Validation Trigger)

CREATE OR REPLACE TRIGGER trg_no_duplicate_name BEFORE INSERT OR UPDATE OF ENAME ON EMP FOR EACH ROW BEGIN IF EXISTS (SELECT 1 FROM EMP WHERE ENAME=:NEW.ENAME) THEN RAISE_APPLICATION_ERROR(-20007,'Duplicate Employee Name not allowed'); END IF; END;

Q26: Trigger to log user activity (who updated, when updated)

CREATE OR REPLACE TRIGGER trg_user_activity AFTER UPDATE ON EMP FOR EACH ROW BEGIN INSERT INTO EMP_ACTIVITY VALUES(USER,SYSDATE,:OLD.EMPNO,'UPDATE'); END;

Q27: Trigger to restrict salary update beyond 20%

CREATE OR REPLACE TRIGGER trg_limit_salary_update BEFORE UPDATE OF SAL ON EMP FOR EACH ROW BEGIN IF :NEW.SAL>:OLD.SAL*1.2 THEN RAISE_APPLICATION_ERROR(-20008,'Salary increase cannot exceed 20%'); END IF; END;

Q28: Trigger that calls a procedure from a package (Trigger + Package Integration)

CREATE OR REPLACE TRIGGER trg_call_pkg AFTER INSERT ON EMP FOR EACH ROW BEGIN audit_pkg.log_insert(:NEW.EMPNO); END;

Q29: Trigger to maintain history table for salary changes

CREATE OR REPLACE TRIGGER trg_salary_history AFTER UPDATE OF SAL ON EMP FOR EACH ROW BEGIN INSERT INTO SAL_HISTORY VALUES(:OLD.EMPNO,:OLD.SAL,:NEW.SAL,SYSDATE); END;

Q30: Compound trigger (advanced Oracle feature)

CREATE OR REPLACE TRIGGER trg_compound FOR UPDATE OF SAL ON EMP COMPOUND TRIGGER v_count NUMBER:=0; BEFORE STATEMENT IS BEGIN DBMS_OUTPUT.PUT_LINE('Before Update Statement'); END BEFORE STATEMENT; AFTER EACH ROW IS BEGIN v_count:=v_count+1; DBMS_OUTPUT.PUT_LINE('Row Updated: '||:NEW.EMPNO); END AFTER EACH ROW; AFTER STATEMENT IS BEGIN DBMS_OUTPUT.PUT_LINE('Total Rows Updated='||v_count); END AFTER STATEMENT; END trg_compound;
