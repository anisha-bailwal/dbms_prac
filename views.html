Views

Q1: Create a view to display employee name, salary, and department number from EMP table

CREATE VIEW emp_view AS SELECT ENAME, SAL, DEPTNO FROM EMP;

Q2: Create a view to display employee name and department name by joining EMP and DEPT tables

CREATE VIEW emp_dept_view AS SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO = D.DEPTNO;

Q3: Create a view to display employees working in department 10

CREATE VIEW dept10_view AS SELECT * FROM EMP WHERE DEPTNO = 10;

Q4: Create a view to display employee name, job, and annual salary

CREATE VIEW annual_sal_view AS SELECT ENAME, JOB, SAL*12 AS Annual_Salary FROM EMP;

Q5: Create a view to display department name and location from DEPT table

CREATE VIEW dept_view AS SELECT DNAME, LOC FROM DEPT;

Q6: Create a view to display employee name, department name, and location

CREATE VIEW emp_dept_loc_view AS SELECT E.ENAME, D.DNAME, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO = D.DEPTNO;

Q7: Create a view to display employees earning more than 3000

CREATE VIEW high_sal_view AS SELECT * FROM EMP WHERE SAL > 3000;

Q8: Create a view to display department-wise total salary

CREATE VIEW dept_sal_view AS SELECT DEPTNO, SUM(SAL) AS Total_Salary FROM EMP GROUP BY DEPTNO;

Q9: Create a view to display only managers

CREATE VIEW manager_view AS SELECT * FROM EMP WHERE JOB = 'MANAGER';

Q10: Create a view with column aliases for employee name and salary

CREATE VIEW alias_view AS SELECT ENAME AS Employee_Name, SAL AS Salary FROM EMP;

Q11: Create a view for employees in department 20 with WITH CHECK OPTION

CREATE VIEW dept20_view AS SELECT * FROM EMP WHERE DEPTNO = 20 WITH CHECK OPTION;

Q32: Create a READ ONLY view showing employee name, department name, and salary

CREATE VIEW emp_dept_sal_readonly AS SELECT E.ENAME, D.DNAME, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WITH READ ONLY;

Q33: Create a complex view to display department name and highest salary in each department

CREATE VIEW dept_high_sal AS SELECT D.DNAME, MAX(E.SAL) AS Highest_Salary FROM EMP E JOIN DEPT D ON E;

Q34: Create a materialized view based on the complex query with aggregate and join

CREATE MATERIALIZED VIEW dept_high_sal_mv AS SELECT D.DNAME, MAX(E.SAL) AS Highest_Salary FROM EMP E JOIN DEPT D ON E.DEPTNO = D.DEPTNO GROUP BY D.DNAME;

Q35: Create a view that hides salary but shows employee name and department name

Q36: Create a complex view to display department name and highest salary in each department
