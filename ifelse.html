PL/SQL:Conditional Statements

Q1: Write a PL/SQL block to check if a number is positive (IF Statement)

DECLARE n NUMBER := 10; BEGIN IF n > 0 THEN DBMS_OUTPUT.PUT_LINE('Positive'); END IF; END;

Q2: Check if an employee's salary is greater than 3000 (IF Statement)

DECLARE v_sal NUMBER := 3500; BEGIN IF v_sal > 3000 THEN DBMS_OUTPUT.PUT_LINE('Salary above 3000'); END IF; END;

Q3: Check if a given number is even (IF Statement)

DECLARE n NUMBER := 8; BEGIN IF MOD(n,2)=0 THEN DBMS_OUTPUT.PUT_LINE('Even'); END IF; END;

Q4: Display a message if today is Monday (IF Statement)

BEGIN IF TO_CHAR(SYSDATE,'DY')='MON' THEN DBMS_OUTPUT.PUT_LINE('Today is Monday'); END IF; END;

Q5: Check if a number is divisible by 5 (IF Statement)

DECLARE n NUMBER := 25; BEGIN IF MOD(n,5)=0 THEN DBMS_OUTPUT.PUT_LINE('Divisible by 5'); END IF; END;

Q6: Check whether a number is even or odd (IF-ELSE)

DECLARE n NUMBER := 7; BEGIN IF MOD(n,2)=0 THEN DBMS_OUTPUT.PUT_LINE('Even'); ELSE DBMS_OUTPUT.PUT_LINE('Odd'); END IF; END;

Q7: Check whether a number is positive or negative (IF-ELSE)

DECLARE n NUMBER := -5; BEGIN IF n >= 0 THEN DBMS_OUTPUT.PUT_LINE('Positive'); ELSE DBMS_OUTPUT.PUT_LINE('Negative'); END IF; END;

Q8: Compare two numbers and display the greater one (IF-ELSE)

DECLARE a NUMBER := 10; b NUMBER := 20; BEGIN IF a > b THEN DBMS_OUTPUT.PUT_LINE('Greater: '||a); ELSE DBMS_OUTPUT.PUT_LINE('Greater: '||b); END IF; END;

Q9: Check whether a student passed or failed (IF-ELSE)

DECLARE marks NUMBER := 35; BEGIN IF marks >= 40 THEN DBMS_OUTPUT.PUT_LINE('Pass'); ELSE DBMS_OUTPUT.PUT_LINE('Fail'); END IF; END;

Q10: Check if salary is above average (IF-ELSE)

DECLARE v_sal NUMBER := 4000; avg_sal NUMBER := 3000; BEGIN IF v_sal > avg_sal THEN DBMS_OUTPUT.PUT_LINE('Above Average'); ELSE DBMS_OUTPUT.PUT_LINE('Below Average'); END IF; END;

Q11: Assign grades using IF-ELSIF-ELSE

DECLARE marks NUMBER := 75; BEGIN IF marks >= 80 THEN DBMS_OUTPUT.PUT_LINE('Grade A'); ELSIF marks >= 60 THEN DBMS_OUTPUT.PUT_LINE('Grade B'); ELSIF marks >= 40 THEN DBMS_OUTPUT.PUT_LINE('Grade C'); ELSE DBMS_OUTPUT.PUT_LINE('Fail'); END IF; END;

Q12: Find the greatest of three numbers (IF-ELSIF-ELSE)

DECLARE a NUMBER := 10; b NUMBER := 20; c NUMBER := 15; BEGIN IF a > b AND a > c THEN DBMS_OUTPUT.PUT_LINE('Greatest: '||a); ELSIF b > c THEN DBMS_OUTPUT.PUT_LINE('Greatest: '||b); ELSE DBMS_OUTPUT.PUT_LINE('Greatest: '||c); END IF; END;

Q13: Check if a number is Positive, Negative, or Zero (IF-ELSIF-ELSE)

DECLARE n NUMBER := 0; BEGIN IF n > 0 THEN DBMS_OUTPUT.PUT_LINE('Positive'); ELSIF n < 0 THEN DBMS_OUTPUT.PUT_LINE('Negative'); ELSE DBMS_OUTPUT.PUT_LINE('Zero'); END IF; END;

Q14: Categorize salary (High/Medium/Low)

DECLARE sal NUMBER := 4500; BEGIN IF sal > 5000 THEN DBMS_OUTPUT.PUT_LINE('High'); ELSIF sal BETWEEN 3000 AND 5000 THEN DBMS_OUTPUT.PUT_LINE('Medium'); ELSE DBMS_OUTPUT.PUT_LINE('Low'); END IF; END;

Q15: Calculate bonus based on job role (IF-ELSIF-ELSE)

DECLARE job VARCHAR2(20) := 'MANAGER'; sal NUMBER := 4000; BEGIN IF job='MANAGER' THEN DBMS_OUTPUT.PUT_LINE('Bonus: '||(sal*0.2)); ELSIF job='CLERK' THEN DBMS_OUTPUT.PUT_LINE('Bonus: '||(sal*0.1)); ELSE DBMS_OUTPUT.PUT_LINE('Bonus: '||(sal*0.05)); END IF; END;

Q16: Display day of week based on number (CASE)

DECLARE d NUMBER := 3; BEGIN CASE d WHEN 1 THEN DBMS_OUTPUT.PUT_LINE('Monday'); WHEN 2 THEN DBMS_OUTPUT.PUT_LINE('Tuesday'); WHEN 3 THEN DBMS_OUTPUT.PUT_LINE('Wednesday'); WHEN 4 THEN DBMS_OUTPUT.PUT_LINE('Thursday'); WHEN 5 THEN DBMS_OUTPUT.PUT_LINE('Friday'); WHEN 6 THEN DBMS_OUTPUT.PUT_LINE('Saturday'); WHEN 7 THEN DBMS_OUTPUT.PUT_LINE('Sunday'); END CASE; END;

Q17: Assign grades using CASE

DECLARE marks NUMBER := 85; BEGIN CASE WHEN marks >= 80 THEN DBMS_OUTPUT.PUT_LINE('Grade A'); WHEN marks >= 60 THEN DBMS_OUTPUT.PUT_LINE('Grade B'); WHEN marks >= 40 THEN DBMS_OUTPUT.PUT_LINE('Grade C'); ELSE DBMS_OUTPUT.PUT_LINE('Fail'); END CASE; END;

Q18: Check whether a number is even or odd (CASE)

DECLARE n NUMBER := 9; BEGIN CASE MOD(n,2) WHEN 0 THEN DBMS_OUTPUT.PUT_LINE('Even'); ELSE DBMS_OUTPUT.PUT_LINE('Odd'); END CASE; END;

Q19: Write a PL/SQL block using CASE to display department name based on department id

Q20: Categorize salary as High/Medium/Low (CASE Statement)

Q21: Print numbers 1–10 but skip number 5 using CONTINUE

BEGIN FOR i IN 1..10 LOOP IF i=5 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q22: Display only odd numbers between 1 and 20 using CONTINUE

BEGIN FOR i IN 1..20 LOOP IF MOD(i,2)=0 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q23: Skip multiples of 3 while printing numbers 1–15 (CONTINUE)

BEGIN FOR i IN 1..15 LOOP IF MOD(i,3)=0 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q24: Skip employees with salary less than 2000 using loop and CONTINUE

DECLARE CURSOR c IS SELECT ENAME,SAL FROM EMP; v_name EMP.ENAME%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c; LOOP FETCH c INTO v_name,v_sal; EXIT WHEN c%NOTFOUND; IF v_sal < 2000 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(v_name||' '||v_sal); END LOOP; CLOSE c; END;

Q25: Exit loop when number reaches 5 (EXIT Statement)

BEGIN FOR i IN 1..10 LOOP IF i=5 THEN EXIT; END IF; DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q26: Print numbers until sum exceeds 50, then exit (EXIT Statement)

DECLARE sum NUMBER := 0; BEGIN FOR i IN 1..20 LOOP sum := sum+i; IF sum > 50 THEN EXIT; END IF; DBMS_OUTPUT.PUT_LINE('Sum='||sum); END LOOP; END;

Q27: Exit loop when a negative number is encountered (EXIT Statement)

DECLARE arr SYS.DBMS_DEBUG_VC2COLL := SYS.DBMS_DEBUG_VC2COLL('10','-5','20'); v NUMBER; BEGIN FOR i IN 1..arr.COUNT LOOP v := TO_NUMBER(arr(i)); IF v < 0 THEN EXIT; END IF; DBMS_OUTPUT.PUT_LINE(v); END LOOP; END;

Q28: Use EXIT WHEN to stop loop when condition is met

DECLARE i NUMBER := 1; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(i); i := i+1; EXIT WHEN i>10; END LOOP; END;

Q29: Use IF-ELSIF and CASE together

DECLARE marks NUMBER := 65; BEGIN IF marks >= 80 THEN DBMS_OUTPUT.PUT_LINE('Grade A'); ELSIF marks >= 60 THEN CASE WHEN marks BETWEEN 60 AND 69 THEN DBMS_OUTPUT.PUT_LINE('Grade B1'); ELSE DBMS_OUTPUT.PUT_LINE('Grade B2'); END CASE; ELSE DBMS_OUTPUT.PUT_LINE('Below 60'); END IF; END;

Q30: Process employee salaries using LOOP, CONTINUE, and EXIT

DECLARE CURSOR c IS SELECT ENAME,SAL FROM EMP; v_name EMP.ENAME%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c; LOOP FETCH c INTO v_name,v_sal; EXIT WHEN c%NOTFOUND; IF v_sal < 2000 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(v_name||' '||v_sal); IF v_sal > 8000 THEN EXIT; END IF; END LOOP; CLOSE c; END;
