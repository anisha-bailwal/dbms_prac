Joins

Q1: Display employee name and department name

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;

Q2: Display employee name, salary, and department location

SELECT E.ENAME, E.SAL, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;

Q3: Display employee name and department name for department 10

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.DEPTNO=10;

Q4: Display employee name and department name where salary > 2000

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.SAL>2000;

Q5: Display employee name, job, and department name

SELECT E.ENAME, E.JOB, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;

Q6: Display employee name and department location for CLERKs

SELECT E.ENAME, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.JOB='CLERK';

Q7: Display employee name and department name working in NEW YORK

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE D.LOC='NEW YORK';

Q8: Display employee name, department number, and department name

SELECT E.ENAME, E.DEPTNO, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;

Q9: Display employees hired after 1981 with department name

SELECT E.ENAME, E.HIREDATE, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.HIREDATE > TO_DATE('01-JAN-1981','DD-MON-YYYY');

Q10: Display employees with commission along with department name

SELECT E.ENAME, E.COMM, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.COMM IS NOT NULL;

Q11: Display employees whose salary is between 1500 and 3000 with department

SELECT E.ENAME, E.SAL, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.SAL BETWEEN 1500 AND 3000;

Q12: Display employees who are the only employee in their department

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.DEPTNO IN (SELECT DEPTNO FROM EMP GROUP BY DEPTNO HAVING COUNT(*)=1);

Q13: Display employees whose department has more than 3 employees

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.DEPTNO IN (SELECT DEPTNO FROM EMP GROUP BY DEPTNO HAVING COUNT(*)>3);

Q14: Display departments where no employee earns more than 2000

SELECT D.DNAME FROM DEPT D WHERE D.DEPTNO NOT IN (SELECT DEPTNO FROM EMP WHERE SAL>2000);

Q15: Display employees who earn more than at least one manager

SELECT ENAME, SAL FROM EMP WHERE SAL > ANY (SELECT SAL FROM EMP WHERE JOB='MANAGER');

Q16: Display employees who share same job and department

SELECT E1.ENAME, E1.JOB, E1.DEPTNO FROM EMP E1 WHERE EXISTS (SELECT 1 FROM EMP E2 WHERE E1.JOB=E2.JOB AND E1.DEPTNO=E2.DEPTNO AND E1.EMPNO<>E2.EMPNO);

Q17: Display employees working in same department as their manager

SELECT E.ENAME, E.DEPTNO FROM EMP E JOIN EMP M ON E.MGR=M.EMPNO WHERE E.DEPTNO=M.DEPTNO;

Q18: Display employees whose manager works in a different department

SELECT E.ENAME, E.DEPTNO, M.DEPTNO AS MGR_DEPT FROM EMP E JOIN EMP M ON E.MGR=M.EMPNO WHERE E.DEPTNO<>M.DEPTNO;

Q19: Display employees who are highest paid in their department

SELECT ENAME, SAL, DEPTNO FROM EMP e WHERE SAL=(SELECT MAX(SAL) FROM EMP WHERE DEPTNO=e.DEPTNO);

Q20: Display employees earning more than their department average

SELECT ENAME, SAL, DEPTNO FROM EMP e WHERE SAL > (SELECT AVG(SAL) FROM EMP WHERE DEPTNO=e.DEPTNO);

Q21: Display department with maximum employees

SELECT DEPTNO FROM EMP GROUP BY DEPTNO HAVING COUNT(*)=(SELECT MAX(COUNT(*)) FROM EMP GROUP BY DEPTNO);

Q22: Display department with highest total salary

SELECT DEPTNO FROM EMP GROUP BY DEPTNO HAVING SUM(SAL)=(SELECT MAX(SUM(SAL)) FROM EMP GROUP BY DEPTNO);

Q23: Display department name and average salary

SELECT D.DNAME, AVG(E.SAL) FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO GROUP BY D.DNAME;

Q24: Display department name and total salary

SELECT D.DNAME, SUM(E.SAL) FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO GROUP BY D.DNAME;

Q25: Display department name and number of employees

SELECT D.DNAME, COUNT(E.EMPNO) FROM DEPT D LEFT JOIN EMP E ON D.DEPTNO=E.DEPTNO GROUP BY D.DNAME;

Q26: Display employees not assigned to any department

SELECT ENAME FROM EMP WHERE DEPTNO IS NULL;

Q27: Display departments with no employees

SELECT D.DNAME FROM DEPT D WHERE D.DEPTNO NOT IN (SELECT DISTINCT DEPTNO FROM EMP);

Q28: Display all departments with employees (even if no employees)

SELECT D.DNAME, E.ENAME FROM DEPT D LEFT JOIN EMP E ON D.DEPTNO=E.DEPTNO;

Q29: Display all employees with department name (even if no department)

SELECT E.ENAME, D.DNAME FROM EMP E LEFT JOIN DEPT D ON E.DEPTNO=D.DEPTNO;

Q30: Display employees whose manager works in department 10

SELECT E.ENAME, M.ENAME AS MANAGER FROM EMP E JOIN EMP M ON E.MGR=M.EMPNO WHERE M.DEPTNO=10;

Q31: Display employees working under the same manager

SELECT E1.ENAME, E2.ENAME AS COLLEAGUE, E1.MGR FROM EMP E1 JOIN EMP E2 ON E1.MGR=E2.MGR WHERE E1.EMPNO<>E2.EMPNO;

Q32: Display employees earning more than their manager

SELECT E.ENAME, E.SAL, M.SAL AS MGR_SAL FROM EMP E JOIN EMP M ON E.MGR=M.EMPNO WHERE E.SAL>M.SAL;

Q33: Display employee name and manager salary

SELECT E.ENAME, M.SAL AS MGR_SAL FROM EMP E JOIN EMP M ON E.MGR=M.EMPNO;

Q34: Display employees who are managers

SELECT DISTINCT M.ENAME FROM EMP M WHERE M.EMPNO IN (SELECT E.MGR FROM EMP E WHERE E.MGR IS NOT NULL);

Q35: Display employees who do not have managers

SELECT ENAME FROM EMP WHERE MGR IS NULL;

Q36: Display employee name and manager name

SELECT E.ENAME, M.ENAME AS MANAGER FROM EMP E LEFT JOIN EMP M ON E.MGR=M.EMPNO;

Q37: Display employees whose department location is DALLAS

SELECT E.ENAME, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE D.LOC='DALLAS';

Q38: Display employees whose department is ACCOUNTING

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE D.DNAME='ACCOUNTING';

Q39: Display employees not working in department 30 with department name

SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.DEPTNO<>30;

Q40: Display employees whose job is MANAGER with department location

SELECT E.ENAME, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.JOB='MANAGER';
