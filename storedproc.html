Stored Functions and Procedures

Q1: Stored procedure to display "Hello Oracle" (Basic)

CREATE OR REPLACE PROCEDURE hello_oracle IS BEGIN DBMS_OUTPUT.PUT_LINE('Hello Oracle'); END;

Q2: Stored procedure to display employee name for a given empno (Procedure + SELECT INTO)

CREATE OR REPLACE PROCEDURE get_emp_name(p_empno NUMBER) IS v_name EMP.ENAME%TYPE; BEGIN SELECT ENAME INTO v_name FROM EMP WHERE EMPNO=p_empno; DBMS_OUTPUT.PUT_LINE('Employee Name: '||v_name); END;

Q3: Stored procedure to print salary of an employee (Procedure + SELECT)

CREATE OR REPLACE PROCEDURE get_emp_salary(p_empno NUMBER) IS v_sal EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=p_empno; DBMS_OUTPUT.PUT_LINE('Salary: '||v_sal); END;

Q4: Stored procedure to display all employees using cursor (Procedure + Cursor)

CREATE OR REPLACE PROCEDURE show_emps IS CURSOR c IS SELECT ENAME FROM EMP; BEGIN FOR rec IN c LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME); END LOOP; END;

Q5: Stored procedure to display department details (Procedure – Basic)

CREATE OR REPLACE PROCEDURE show_depts IS CURSOR c IS SELECT DNAME,LOC FROM DEPT; BEGIN FOR rec IN c LOOP DBMS_OUTPUT.PUT_LINE(rec.DNAME||' '||rec.LOC); END LOOP; END;

Q6: Procedure with IN parameter to display employee details

CREATE OR REPLACE PROCEDURE emp_details(p_empno NUMBER) IS v_name EMP.ENAME%TYPE; v_job EMP.JOB%TYPE; BEGIN SELECT ENAME,JOB INTO v_name,v_job FROM EMP WHERE EMPNO=p_empno; DBMS_OUTPUT.PUT_LINE(v_name||' '||v_job); END;

Q7: Procedure with IN parameter to display employees of a department

CREATE OR REPLACE PROCEDURE dept_emps(p_deptno NUMBER) IS BEGIN FOR rec IN (SELECT ENAME FROM EMP WHERE DEPTNO=p_deptno) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME); END LOOP; END;

Q8: Procedure with IN parameter to display employees earning more than given salary

CREATE OR REPLACE PROCEDURE high_salary(p_sal NUMBER) IS BEGIN FOR rec IN (SELECT ENAME,SAL FROM EMP WHERE SAL>p_sal) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); END LOOP; END;

Q9: Procedure with OUT parameter to return employee salary

CREATE OR REPLACE PROCEDURE get_salary(p_empno NUMBER, p_sal OUT NUMBER) IS BEGIN SELECT SAL INTO p_sal FROM EMP WHERE EMPNO=p_empno; END;

Q10: Procedure with IN OUT parameter to update salary and return new salary

CREATE OR REPLACE PROCEDURE update_salary(p_empno NUMBER, p_sal IN OUT NUMBER) IS BEGIN UPDATE EMP SET SAL=p_sal WHERE EMPNO=p_empno; SELECT SAL INTO p_sal FROM EMP WHERE EMPNO=p_empno; END;

Q11: Procedure to insert a new employee record (Procedure + INSERT)

CREATE OR REPLACE PROCEDURE add_emp(p_empno NUMBER, p_name VARCHAR2, p_job VARCHAR2, p_sal NUMBER, p_deptno NUMBER) IS BEGIN INSERT INTO EMP(EMPNO,ENAME,JOB,SAL,DEPTNO) VALUES(p_empno,p_name,p_job,p_sal,p_deptno); END;

Q12: Procedure to update employee salary based on empno (Procedure + UPDATE)

CREATE OR REPLACE PROCEDURE update_emp_salary(p_empno NUMBER, p_sal NUMBER) IS BEGIN UPDATE EMP SET SAL=p_sal WHERE EMPNO=p_empno; END;

Q13: Procedure to delete an employee record (Procedure + DELETE)

CREATE OR REPLACE PROCEDURE delete_emp(p_empno NUMBER) IS BEGIN DELETE FROM EMP WHERE EMPNO=p_empno; END;

Q14: Procedure to increase salary of all employees by 10% (Procedure + UPDATE)

CREATE OR REPLACE PROCEDURE inc_salary IS BEGIN UPDATE EMP SET SAL=SAL*1.1; END;

Q15: Procedure to transfer employees from one department to another (Procedure + Business Logic)

CREATE OR REPLACE PROCEDURE transfer_emp(p_from NUMBER, p_to NUMBER) IS BEGIN UPDATE EMP SET DEPTNO=p_to WHERE DEPTNO=p_from; END;

Q16: Function to return square of a number (Function – Basic)

CREATE OR REPLACE FUNCTION square_num(p_num NUMBER) RETURN NUMBER IS BEGIN RETURN p_num*p_num; END;

Q17: Function to return employee salary based on empno (Function + SELECT INTO)

CREATE OR REPLACE FUNCTION get_salary(p_empno NUMBER) RETURN NUMBER IS v_sal EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=p_empno; RETURN v_sal; END;

Q18: Function to return total number of employees (Function + Aggregate)

CREATE OR REPLACE FUNCTION total_emps RETURN NUMBER IS v_cnt NUMBER; BEGIN SELECT COUNT(*) INTO v_cnt FROM EMP; RETURN v_cnt; END;

Q19: Function to return maximum salary from EMP table (Function + Aggregate)

CREATE OR REPLACE FUNCTION max_salary RETURN NUMBER IS v_sal NUMBER; BEGIN SELECT MAX(SAL) INTO v_sal FROM EMP; RETURN v_sal; END;

Q20: Function to return department name based on deptno (Function)

CREATE OR REPLACE FUNCTION get_dept_name(p_deptno NUMBER) RETURN VARCHAR2 IS v_name DEPT.DNAME%TYPE; BEGIN SELECT DNAME INTO v_name FROM DEPT WHERE DEPTNO=p_deptno; RETURN v_name; END;

Q21: Function to calculate bonus (10% of salary) and return it (Function + Calculation)

CREATE OR REPLACE FUNCTION calc_bonus(p_sal NUMBER) RETURN NUMBER IS BEGIN RETURN p_sal*0.1; END;

Q22: Function to return grade based on salary (Function + IF-ELSIF)

CREATE OR REPLACE FUNCTION grade(p_sal NUMBER) RETURN VARCHAR2 IS BEGIN IF p_sal>5000 THEN RETURN 'A'; ELSIF p_sal>3000 THEN RETURN 'B'; ELSE RETURN 'C'; END IF; END;

Q23: Function to check whether a number is even or odd (Function + Logic)

CREATE OR REPLACE FUNCTION check_num(p_num NUMBER) RETURN VARCHAR2 IS BEGIN IF MOD(p_num,2)=0 THEN RETURN 'EVEN'; ELSE RETURN 'ODD'; END IF; END;

Q24: Function to return annual salary of an employee (Function + Calculation)

CREATE OR REPLACE FUNCTION annual_salary(p_empno NUMBER) RETURN NUMBER IS v_sal EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=p_empno; RETURN v_sal*12; END;

Q25: Function to return employee experience (based on hiredate) (Function + Date Logic)

CREATE OR REPLACE FUNCTION emp_experience(p_empno NUMBER) RETURN NUMBER IS v_date EMP.HIREDATE%TYPE; BEGIN SELECT HIREDATE INTO v_date FROM EMP WHERE EMPNO=p_empno; RETURN TRUNC(MONTHS_BETWEEN(SYSDATE,v_date)/12); END;

Q26: Procedure that calls a function to calculate bonus and display result (Procedure calling Function)

CREATE OR REPLACE PROCEDURE show_bonus(p_empno NUMBER) IS v_sal EMP.SAL%TYPE; v_bonus NUMBER; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=p_empno; v_bonus := calc_bonus(v_sal); DBMS_OUTPUT.PUT_LINE('Bonus='||v_bonus); END;

Q27: Function and use it inside SQL query (Function in SQL)

CREATE OR REPLACE FUNCTION annual_sal(p_sal NUMBER) RETURN NUMBER IS BEGIN RETURN p_sal*12; END; -- Usage: -- SELECT ENAME, annual_sal(SAL) FROM EMP;

Q28: Procedure that uses function to update salary (Procedure + Function)

CREATE OR REPLACE PROCEDURE update_with_bonus(p_empno NUMBER) IS v_sal EMP.SAL%TYPE; v_bonus NUMBER; BEGIN SELECT SAL INTO v_sal FROM EMP WHERE EMPNO=p_empno; v_bonus := calc_bonus(v_sal); UPDATE EMP SET SAL=SAL+v_bonus WHERE EMPNO=p_empno; END;

Q29: Function that returns highest salary and use it in procedure (Integration)

CREATE OR REPLACE FUNCTION highest_sal RETURN NUMBER IS v_sal NUMBER; BEGIN SELECT MAX(SAL) INTO v_sal FROM EMP; RETURN v_sal; END; / CREATE OR REPLACE PROCEDURE show_highest IS BEGIN DBMS_OUTPUT.PUT_LINE('Highest Salary='||highest_sal); END;

Q30: Stored procedure to display top 5 highest paid employees (Procedure + Advanced Query)

CREATE OR REPLACE PROCEDURE top5_emps IS BEGIN FOR rec IN (SELECT ENAME,SAL FROM EMP ORDER BY SAL DESC FETCH FIRST 5 ROWS ONLY) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); END LOOP; END;

Q31: Stored function to return nth highest salary (Function + Logic)

CREATE OR REPLACE FUNCTION nth_highest(n NUMBER) RETURN NUMBER IS v_sal NUMBER; BEGIN SELECT SAL INTO v_sal FROM ( SELECT DISTINCT SAL FROM EMP ORDER BY SAL DESC ) WHERE ROWNUM=n; RETURN v_sal; END;

Q32: Procedure to implement transaction (COMMIT / ROLLBACK)

CREATE OR REPLACE PROCEDURE transfer_emp_txn(p_from NUMBER, p_to NUMBER) IS BEGIN UPDATE EMP SET DEPTNO=p_to WHERE DEPTNO=p_from; COMMIT; EXCEPTION WHEN OTHERS THEN ROLLBACK; END;

Q33: Function that returns department-wise total salary (Function + GROUP BY)

CREATE OR REPLACE FUNCTION dept_total_sal(p_deptno NUMBER) RETURN NUMBER IS v_total NUMBER; BEGIN SELECT SUM(SAL) INTO v_total FROM EMP WHERE DEPTNO=p_deptno; RETURN v_total; END;
