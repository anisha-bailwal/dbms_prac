Set Operators

Q1: Display all department numbers from both EMP and DEPT tables using UNION

SELECT DEPTNO FROM EMP UNION SELECT DEPTNO FROM DEPT;

Q2: Display all department numbers from EMP and DEPT including duplicates using UNION ALL

SELECT DEPTNO FROM EMP UNION ALL SELECT DEPTNO FROM DEPT;

Q3: Display department numbers present in both EMP and DEPT using INTERSECT

SELECT DEPTNO FROM EMP INTERSECT SELECT DEPTNO FROM DEPT;

Q4: Display department numbers present in EMP but not in DEPT using MINUS

SELECT DEPTNO FROM EMP MINUS SELECT DEPTNO FROM DEPT;

Q5: Display department numbers present in DEPT but not assigned to any employee

SELECT DEPTNO FROM DEPT MINUS SELECT DEPTNO FROM EMP;

Q6: Display employee names and department names in a single column using UNION

SELECT ENAME AS Name FROM EMP UNION SELECT DNAME AS Name FROM DEPT;

Q7: Display all employee names and department names, including duplicates

SELECT ENAME AS Name FROM EMP UNION ALL SELECT DNAME AS Name FROM DEPT;

Q8: Display jobs that exist in EMP and also as department names in DEPT using INTERSECT

SELECT JOB FROM EMP INTERSECT SELECT DNAME FROM DEPT;

Q9: Display employee names that are not department names

SELECT ENAME FROM EMP MINUS SELECT DNAME FROM DEPT;

Q10: Display department locations not used by any employee using MINUS

SELECT LOC FROM DEPT MINUS SELECT LOC FROM EMP;

Q11: Display employee names from department 10 and department 20 using UNION

SELECT ENAME FROM EMP WHERE DEPTNO=10 UNION SELECT ENAME FROM EMP WHERE DEPTNO=20;

Q12: Display employee names from department 10 and department 20 including duplicates

SELECT ENAME FROM EMP WHERE DEPTNO=10 UNION ALL SELECT ENAME FROM EMP WHERE DEPTNO=20;

Q13: Display employees who work in departments that exist in DEPT table using INTERSECT

SELECT ENAME FROM EMP INTERSECT SELECT ENAME FROM EMP WHERE DEPTNO IN (SELECT DEPTNO FROM DEPT);

Q14: Display employees whose department does not exist in DEPT table using MINUS

SELECT ENAME FROM EMP MINUS SELECT ENAME FROM EMP WHERE DEPTNO IN (SELECT DEPTNO FROM DEPT);

Q15: Display department numbers that have employees and also exist in DEPT table

SELECT DEPTNO FROM EMP INTERSECT SELECT DEPTNO FROM DEPT;

Q16: Display all unique job roles from EMP and all unique department names from DEPT in one result

SELECT JOB AS Value FROM EMP UNION SELECT DNAME AS Value FROM DEPT;

Q17: Display common values between employee job and department location

SELECT JOB FROM EMP INTERSECT SELECT LOC FROM DEPT;

Q18: Display employee names who are not assigned to any valid department

SELECT ENAME FROM EMP WHERE DEPTNO NOT IN (SELECT DEPTNO FROM DEPT);

Q19: Combine employees earning > 2000 and employees from department 10 using UNION

SELECT ENAME FROM EMP WHERE SAL > 2000 UNION SELECT ENAME FROM EMP WHERE DEPTNO=10;

Q20: Display employees in department 10 but not in department 20 using MINUS

SELECT ENAME FROM EMP WHERE DEPTNO=10 MINUS SELECT ENAME FROM EMP WHERE DEPTNO=20;

Q21: Display department numbers that exist in EMP and DEPT but exclude department 10

SELECT DEPTNO FROM EMP INTERSECT SELECT DEPTNO FROM DEPT MINUS SELECT 10 FROM DUAL;

Q22: Display all employee names and department names in a single result sorted alphabetically

SELECT ENAME AS Name FROM EMP UNION SELECT DNAME AS Name FROM DEPT ORDER BY Name;

Q23: Display department numbers that appear in EMP more than once using UNION ALL

SELECT DEPTNO FROM EMP UNION ALL SELECT DEPTNO FROM EMP;

Q24: Display employees from departments located in NEW YORK and DALLAS using UNION

SELECT ENAME FROM EMP WHERE DEPTNO IN (SELECT DEPTNO FROM DEPT WHERE LOC='NEW YORK') UNION SELECT ENAME FROM EMP WHERE DEPTNO IN (SELECT DEPTNO FROM DEPT WHERE LOC='DALLAS');

Q25: Combine employees with salary > 3000 and employees with job = 'MANAGER' — remove duplicates

SELECT ENAME FROM EMP WHERE SAL > 3000 UNION SELECT ENAME FROM EMP WHERE JOB='MANAGER';
