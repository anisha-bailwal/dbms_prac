Q1: Insert a record into STUDENT table: SID=101, SNAME='RAHUL', AGE=20, COURSE='MCA'

INSERT INTO STUDENT (SID, SNAME, AGE, COURSE) VALUES (101, 'RAHUL', 20, 'MCA');

Q2: Insert a record into EMPLOYEE table: EID=201, ENAME='ANITA', SALARY=45000, DEPTNO=10

INSERT INTO EMPLOYEE (EID, ENAME, SALARY, DEPTNO) VALUES (201, 'ANITA', 45000, 10);

Q3: Insert a new department into DEPT table: DEPTNO=50, DNAME='TRAINING', LOC='DELHI'

INSERT INTO DEPT (DEPTNO, DNAME, LOC) VALUES (50, 'TRAINING', 'DELHI');

Q4: Insert a record into BOOK table with only BOOK_ID, TITLE, and PRICE

INSERT INTO BOOK (BOOK_ID, TITLE, PRICE) VALUES (1, 'SQL Basics', 500);

Q6: Insert data into EMP_COPY table from the EMP table

INSERT INTO EMP_COPY SELECT * FROM EMP;

Q7: Insert employees from EMP whose salary > 3000 into EMP_HIGH_SAL

INSERT INTO EMP_HIGH_SAL SELECT * FROM EMP WHERE SAL > 3000;

Q8: Insert employees hired after 1981 into EMP_NEW_HIRE

INSERT INTO EMP_NEW_HIRE SELECT * FROM EMP WHERE HIREDATE > TO_DATE('01-JAN-1981','DD-MON-YYYY');

Q9: Insert department numbers and names from DEPT into DEPT_BACKUP

INSERT INTO DEPT_BACKUP (DEPTNO, DNAME) SELECT DEPTNO, DNAME FROM DEPT;

Q10: Insert employee names and salaries into EMP_SALARY_RECORD

INSERT INTO EMP_SALARY_RECORD (ENAME, SAL) SELECT ENAME, SAL FROM EMP;

Q11: Update the salary of employee SMITH to 2000

UPDATE EMP SET SAL = 2000 WHERE ENAME = 'SMITH';

Q12: Increase salary of all employees by 10%

UPDATE EMP SET SAL = SAL * 1.10;

Q13: Update the department number of employee ALLEN to 20

UPDATE EMP SET DEPTNO = 20 WHERE ENAME = 'ALLEN';

Q14: Update commission of SALESMAN to 500

UPDATE EMP SET COMM = 500 WHERE JOB = 'SALESMAN';

Q15: Update job of employee MILLER to MANAGER

UPDATE EMP SET JOB = 'MANAGER' WHERE ENAME = 'MILLER';

Q16: Increase salary of employees in department 10 by 20%

UPDATE EMP SET SAL = SAL * 1.20 WHERE DEPTNO = 10;

Q17: Update salary of employees whose job is CLERK to 1500

UPDATE EMP SET SAL = 1500 WHERE JOB = 'CLERK';

Q18: Update location of department 30 to MUMBAI

UPDATE DEPT SET LOC = 'MUMBAI' WHERE DEPTNO = 30;

Q19: Update commission of employees who have NULL commission to 100

UPDATE EMP SET COMM = 100 WHERE COMM IS NULL;

Q20: Update salary of employees who earn less than average salary

UPDATE EMP SET SAL = SAL + 500 WHERE SAL < (SELECT AVG(SAL) FROM EMP);

Q21: Delete employee SMITH from EMP table

DELETE FROM EMP WHERE ENAME = 'SMITH';

Q22: Delete all employees working in department 30

DELETE FROM EMP WHERE DEPTNO = 30;

Q23: Delete employees whose salary is less than 1000

DELETE FROM EMP WHERE SAL < 1000;

Q24: Delete employees hired before 1981

DELETE FROM EMP WHERE HIREDATE < TO_DATE('01-JAN-1981','DD-MON-YYYY');

Q25: Delete employees who do not receive commission

DELETE FROM EMP WHERE COMM IS NULL;

Q26: Delete departments located in CHICAGO

DELETE FROM DEPT WHERE LOC = 'CHICAGO';

Q27: Delete employees whose job is CLERK

DELETE FROM EMP WHERE JOB = 'CLERK';

Q28: Delete employees earning the minimum salary

DELETE FROM EMP WHERE SAL = (SELECT MIN(SAL) FROM EMP);

Q29: Delete employees working in departments located in NEW YORK

DELETE FROM EMP WHERE DEPTNO IN (SELECT DEPTNO FROM DEPT WHERE LOC = 'NEW YORK');
