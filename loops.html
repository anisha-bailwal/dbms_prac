Iterative Control Statements

Q1: Print numbers from 1 to 5 using SIMPLE LOOP

DECLARE i NUMBER := 1; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(i); i := i+1; EXIT WHEN i>5; END LOOP; END;

Q2: Display numbers from 10 to 1 in reverse order using SIMPLE LOOP

DECLARE i NUMBER := 10; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(i); i := i-1; EXIT WHEN i<1; END LOOP; END;

Q3: Calculate sum of numbers from 1 to 10 using SIMPLE LOOP

DECLARE i NUMBER := 1; s NUMBER := 0; BEGIN LOOP s := s+i; i := i+1; EXIT WHEN i>10; END LOOP; DBMS_OUTPUT.PUT_LINE('Sum='||s); END;

Q4: Print multiplication table of a number using SIMPLE LOOP

DECLARE n NUMBER := 5; i NUMBER := 1; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(n||' x '||i||' = '||(n*i)); i := i+1; EXIT WHEN i>10; END LOOP; END;

Q5: Print even numbers between 1 and 20 using SIMPLE LOOP

DECLARE i NUMBER := 2; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(i); i := i+2; EXIT WHEN i>20; END LOOP; END;

Q6: Exit loop when number becomes greater than 50 (SIMPLE LOOP + EXIT)

DECLARE i NUMBER := 1; BEGIN LOOP DBMS_OUTPUT.PUT_LINE(i); i := i+5; EXIT WHEN i>50; END LOOP; END;

Q7: Print numbers from 1 to 10 using WHILE LOOP

DECLARE i NUMBER := 1; BEGIN WHILE i<=10 LOOP DBMS_OUTPUT.PUT_LINE(i); i := i+1; END LOOP; END;

Q8: Calculate factorial of a number using WHILE LOOP

DECLARE n NUMBER := 5; fact NUMBER := 1; BEGIN WHILE n>0 LOOP fact := fact*n; n := n-1; END LOOP; DBMS_OUTPUT.PUT_LINE('Factorial='||fact); END;

Q9: Display sum of digits of a number using WHILE LOOP

DECLARE n NUMBER := 1234; sum NUMBER := 0; BEGIN WHILE n>0 LOOP sum := sum+MOD(n,10); n := TRUNC(n/10); END LOOP; DBMS_OUTPUT.PUT_LINE('Sum of digits='||sum); END;

Q10: Reverse a number using WHILE LOOP

DECLARE n NUMBER := 1234; rev NUMBER := 0; BEGIN WHILE n>0 LOOP rev := rev*10+MOD(n,10); n := TRUNC(n/10); END LOOP; DBMS_OUTPUT.PUT_LINE('Reverse='||rev); END;

Q11: Check whether a number is palindrome using WHILE LOOP

DECLARE n NUMBER := 121; temp NUMBER := n; rev NUMBER := 0; BEGIN WHILE temp>0 LOOP rev := rev*10+MOD(temp,10); temp := TRUNC(temp/10); END LOOP; IF rev=n THEN DBMS_OUTPUT.PUT_LINE('Palindrome'); ELSE DBMS_OUTPUT.PUT_LINE('Not Palindrome'); END IF; END;

Q12: Print Fibonacci series up to N terms using WHILE LOOP

DECLARE n NUMBER := 10; a NUMBER := 0; b NUMBER := 1; i NUMBER := 1; BEGIN WHILE i<=n LOOP DBMS_OUTPUT.PUT_LINE(a); a := a+b; b := a-b; i := i+1; END LOOP; END;

Q13: Print numbers from 1 to 10 using FOR LOOP

BEGIN FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q14: Print numbers in reverse order using FOR LOOP

BEGIN FOR i IN REVERSE 1..10 LOOP DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q15: Calculate sum of first 10 natural numbers using FOR LOOP

DECLARE s NUMBER := 0; BEGIN FOR i IN 1..10 LOOP s := s+i; END LOOP; DBMS_OUTPUT.PUT_LINE('Sum='||s); END;

Q16: Display multiplication table of 5 using FOR LOOP

BEGIN FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE('5 x '||i||' = '||(5*i)); END LOOP; END;

Q17: Print squares of numbers from 1 to 10 using FOR LOOP

BEGIN FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE('Square of '||i||' = '||(i*i)); END LOOP; END;

Q18: Display all odd numbers between 1 and 50 using FOR LOOP

BEGIN FOR i IN 1..50 LOOP IF MOD(i,2)<>0 THEN DBMS_OUTPUT.PUT_LINE(i); END IF; END LOOP; END;

Q19: Print a * ** *** **** pattern using nested loops

BEGIN FOR i IN 1..4 LOOP FOR j IN 1..i LOOP DBMS_OUTPUT.PUT('*'); END LOOP; DBMS_OUTPUT.PUT_LINE(''); END LOOP; END;

Q20: Print multiplication tables from 1 to 5 using nested loops

BEGIN FOR n IN 1..5 LOOP FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE(n||' x '||i||' = '||(n*i)); END LOOP; DBMS_OUTPUT.PUT_LINE('---'); END LOOP; END;

Q21: Display number pattern: 1 / 12 / 123 / 1234 using nested loops

BEGIN FOR i IN 1..4 LOOP FOR j IN 1..i LOOP DBMS_OUTPUT.PUT(j); END LOOP; DBMS_OUTPUT.PUT_LINE(''); END LOOP; END;

Q22: Print numbers from 1 to 20 but skip multiples of 4 (LOOP + CONTINUE)

BEGIN FOR i IN 1..20 LOOP IF MOD(i,4)=0 THEN CONTINUE; END IF; DBMS_OUTPUT.PUT_LINE(i); END LOOP; END;

Q23: Stop printing numbers once the sum exceeds 100 (LOOP + EXIT)

DECLARE sum NUMBER := 0; BEGIN FOR i IN 1..50 LOOP sum := sum+i; DBMS_OUTPUT.PUT_LINE('Sum='||sum); IF sum>100 THEN EXIT; END IF; END LOOP; END;

Q24: Print numbers and exit when number divisible by 7 is found (LOOP + EXIT)

BEGIN FOR i IN 1..20 LOOP DBMS_OUTPUT.PUT_LINE(i); IF MOD(i,7)=0 THEN EXIT; END IF; END LOOP; END;

Q25: Loop through employee IDs from EMP table and display them (FOR LOOP + Table Data)

DECLARE CURSOR c IS SELECT EMPNO FROM EMP; v_id EMP.EMPNO%TYPE; BEGIN FOR rec IN c LOOP DBMS_OUTPUT.PUT_LINE('Emp ID: '||rec.EMPNO); END LOOP; END;

Q26: Increase salary of employees by 10% using loop (LOOP + UPDATE)

DECLARE CURSOR c IS SELECT EMPNO,SAL FROM EMP; v_id EMP.EMPNO%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c; LOOP FETCH c INTO v_id,v_sal; EXIT WHEN c%NOTFOUND; UPDATE EMP SET SAL=SAL*1.1 WHERE EMPNO=v_id; END LOOP; CLOSE c; DBMS_OUTPUT.PUT_LINE('Salaries Updated'); END;

Q27: Display employees whose salary > 3000 using loop (LOOP + CONDITION)

DECLARE CURSOR c IS SELECT ENAME,SAL FROM EMP; v_name EMP.ENAME%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c; LOOP FETCH c INTO v_name,v_sal; EXIT WHEN c%NOTFOUND; IF v_sal>3000 THEN DBMS_OUTPUT.PUT_LINE(v_name||' '||v_sal); END IF; END LOOP; CLOSE c; END;

Q28: Count total number of employees using loop (LOOP + COUNT)

DECLARE CURSOR c IS SELECT EMPNO FROM EMP; v_id EMP.EMPNO%TYPE; cnt NUMBER := 0; BEGIN OPEN c; LOOP FETCH c INTO v_id; EXIT WHEN c%NOTFOUND; cnt := cnt+1; END LOOP; CLOSE c; DBMS_OUTPUT.PUT_LINE('Total Employees='||cnt); END;

Q29: Use all three loops (SIMPLE, WHILE, FOR) in one program

DECLARE i NUMBER := 1; j NUMBER := 1; BEGIN -- SIMPLE LOOP LOOP DBMS_OUTPUT.PUT_LINE('Simple:'||i); i := i+1; EXIT WHEN i>3; END LOOP; -- WHILE LOOP WHILE j<=3 LOOP DBMS_OUTPUT.PUT_LINE('While:'||j); j := j+1; END LOOP; -- FOR LOOP FOR k IN 1..3 LOOP DBMS_OUTPUT.PUT_LINE('For:'||k); END LOOP; END;

Q30: Generate prime numbers between 1 and 50 (LOOP + LOGIC)

DECLARE flag BOOLEAN; BEGIN FOR n IN 2..50 LOOP flag := TRUE; FOR i IN 2..TRUNC(SQRT(n)) LOOP IF MOD(n,i)=0 THEN flag := FALSE; EXIT; END IF; END LOOP; IF flag THEN DBMS_OUTPUT.PUT_LINE(n); END IF; END LOOP; END;

Q31: Simulate ATM withdrawal attempts — 3 attempts using loop

DECLARE pin NUMBER := 1234; entered NUMBER; attempts NUMBER := 0; BEGIN LOOP entered := 1234; -- simulate input attempts := attempts+1; IF entered=pin THEN DBMS_OUTPUT.PUT_LINE('Access Granted'); EXIT; ELSE DBMS_OUTPUT.PUT_LINE('Wrong PIN'); END IF; EXIT WHEN attempts=3; END LOOP; END;

Q32: Calculate sum of even and odd numbers separately using loop

DECLARE even_sum NUMBER := 0; odd_sum NUMBER := 0; BEGIN FOR i IN 1..20 LOOP IF MOD(i,2)=0 THEN even_sum := even_sum+i; ELSE odd_sum := odd_sum+i; END IF; END LOOP; DBMS_OUTPUT.PUT_LINE('Even Sum='||even_sum); DBMS_OUTPUT.PUT_LINE('Odd Sum='||odd_sum); END;
