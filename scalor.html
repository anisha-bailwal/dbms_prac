Scalor functions

Q1: Display employee name, hire date, current date, and number of days worked

SELECT ENAME, HIREDATE, SYSDATE, SYSDATE - HIREDATE AS Days_Worked FROM EMP;

Q2: Display employee name and hire date in format: Monday, January 01, 2024

SELECT ENAME, TO_CHAR(HIREDATE,'Day, Month DD, YYYY') AS Hire_Date FROM EMP;

Q3: Find employees who joined in the same month as current system date

SELECT ENAME FROM EMP WHERE TO_CHAR(HIREDATE,'MM') = TO_CHAR(SYSDATE,'MM');

Q4: Display employee name and the next Monday after their hire date

SELECT ENAME, NEXT_DAY(HIREDATE,'MONDAY') AS Next_Monday FROM EMP;

Q5: Display employee name and the last day of their joining month

SELECT ENAME, LAST_DAY(HIREDATE) AS Last_Day FROM EMP;

Q6: Find employees who have completed more than 10 years in the company

SELECT ENAME FROM EMP WHERE MONTHS_BETWEEN(SYSDATE,HIREDATE)/12 > 10;

Q7: Display employee name and number of months worked

SELECT ENAME, MONTHS_BETWEEN(SYSDATE,HIREDATE) AS Months_Worked FROM EMP;

Q8: Display employee name, hire date, and hire date after adding 3 years and 6 months

SELECT ENAME, HIREDATE, ADD_MONTHS(HIREDATE,42) AS New_Date FROM EMP;

Q9: Find employees hired between two given dates

SELECT ENAME FROM EMP WHERE HIREDATE BETWEEN TO_DATE('01-JAN-1981','DD-MON-YYYY') AND TO_DATE('31-DEC-1981','DD-MON-YYYY');

Q10: Display difference in years between SYSDATE and hire date - rounded

Q52: Replace NULL department name with 'No Department'

Q53: Display employees whose commission is NULL

Q54: Use DECODE to categorize salary as Low/Medium/High

SELECT ENAME, DECODE(TRUE, SAL<2000,'Low', SAL BETWEEN 2000 AND 4000,'Medium', 'High') AS Category FROM EMP;

Q55: Use CASE to categorize employees based on experience

SELECT ENAME, CASE WHEN MONTHS_BETWEEN(SYSDATE,HIREDATE)/12 < 5 THEN 'Junior' WHEN MONTHS_BETWEEN(SYSDATE,HIREDATE)/12 BETWEEN 5 AND 10 THEN 'Mid-Level' ELSE 'Senior' END AS Experience FROM EMP;
