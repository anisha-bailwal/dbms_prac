Cursors

Q1: Declare a cursor to fetch all employees and display names (Explicit Cursor)

DECLARE CURSOR c_emp IS SELECT ENAME FROM EMP; v_name EMP.ENAME%TYPE; BEGIN OPEN c_emp; LOOP FETCH c_emp INTO v_name; EXIT WHEN c_emp%NOTFOUND; DBMS_OUTPUT.PUT_LINE(v_name); END LOOP; CLOSE c_emp; END;

Q2: Fetch employee name and salary using cursor (Explicit Cursor)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; v_name EMP.ENAME%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c_emp; LOOP FETCH c_emp INTO v_name,v_sal; EXIT WHEN c_emp%NOTFOUND; DBMS_OUTPUT.PUT_LINE(v_name||' '||v_sal); END LOOP; CLOSE c_emp; END;

Q3: Display employees whose salary > 3000 using cursor (Explicit Cursor + Condition)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; v_name EMP.ENAME%TYPE; v_sal EMP.SAL%TYPE; BEGIN OPEN c_emp; LOOP FETCH c_emp INTO v_name,v_sal; EXIT WHEN c_emp%NOTFOUND; IF v_sal>3000 THEN DBMS_OUTPUT.PUT_LINE(v_name||' '||v_sal); END IF; END LOOP; CLOSE c_emp; END;

Q4: Count number of employees using cursor (Explicit Cursor)

DECLARE CURSOR c_emp IS SELECT ENAME FROM EMP; v_name EMP.ENAME%TYPE; cnt NUMBER:=0; BEGIN OPEN c_emp; LOOP FETCH c_emp INTO v_name; EXIT WHEN c_emp%NOTFOUND; cnt:=cnt+1; END LOOP; CLOSE c_emp; DBMS_OUTPUT.PUT_LINE('Total Employees='||cnt); END;

Q5: Calculate total salary of all employees using cursor (Explicit Cursor + Aggregate)

DECLARE CURSOR c_emp IS SELECT SAL FROM EMP; v_sal EMP.SAL%TYPE; total NUMBER:=0; BEGIN OPEN c_emp; LOOP FETCH c_emp INTO v_sal; EXIT WHEN c_emp%NOTFOUND; total:=total+v_sal; END LOOP; CLOSE c_emp; DBMS_OUTPUT.PUT_LINE('Total Salary='||total); END;

Q6: Fetch records one by one using OPEN, FETCH, CLOSE (Cursor Lifecycle)

DECLARE CURSOR c_emp IS SELECT ENAME FROM EMP; v_name EMP.ENAME%TYPE; BEGIN OPEN c_emp; FETCH c_emp INTO v_name; WHILE c_emp%FOUND LOOP DBMS_OUTPUT.PUT_LINE(v_name); FETCH c_emp INTO v_name; END LOOP; CLOSE c_emp; END;

Q7: Display all employee records using cursor FOR LOOP

DECLARE CURSOR c_emp IS SELECT * FROM EMP; BEGIN FOR rec IN c_emp LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL||' '||rec.JOB); END LOOP; END;

Q8: Display employee names and jobs using cursor FOR LOOP

DECLARE CURSOR c_emp IS SELECT ENAME,JOB FROM EMP; BEGIN FOR rec IN c_emp LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' - '||rec.JOB); END LOOP; END;

Q9: Display employees from department 10 using cursor FOR LOOP

DECLARE CURSOR c_emp IS SELECT ENAME,DEPTNO FROM EMP WHERE DEPTNO=10; BEGIN FOR rec IN c_emp LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.DEPTNO); END LOOP; END;

Q10: Calculate total salary using cursor FOR LOOP

DECLARE CURSOR c_emp IS SELECT SAL FROM EMP; total NUMBER:=0; BEGIN FOR rec IN c_emp LOOP total:=total+rec.SAL; END LOOP; DBMS_OUTPUT.PUT_LINE('Total Salary='||total); END;

Q11: Display employees with salary greater than average using cursor FOR LOOP

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; avg_sal NUMBER; BEGIN SELECT AVG(SAL) INTO avg_sal FROM EMP; FOR rec IN c_emp LOOP IF rec.SAL>avg_sal THEN DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); END IF; END LOOP; END;

Q12: Parameterized cursor to display employees of a department

DECLARE CURSOR c_emp(dno NUMBER) IS SELECT ENAME FROM EMP WHERE DEPTNO=dno; BEGIN FOR rec IN c_emp(10) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME); END LOOP; END;

Q15: Parameterized cursor to calculate total salary of a department

DECLARE CURSOR c_emp(dno NUMBER) IS SELECT SAL FROM EMP WHERE DEPTNO=dno; total NUMBER:=0; BEGIN FOR rec IN c_emp(20) LOOP total:=total+rec.SAL; END LOOP; DBMS_OUTPUT.PUT_LINE('Total Salary='||total); END;

Q16: Pass multiple parameters (deptno, job) to cursor

DECLARE CURSOR c_emp(dno NUMBER, j VARCHAR2) IS SELECT ENAME,JOB FROM EMP WHERE DEPTNO=dno AND JOB=j; BEGIN FOR rec IN c_emp(30,'SALESMAN') LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.JOB); END LOOP; END;

Q17: Update salary of employees using cursor FOR UPDATE

DECLARE CURSOR c_emp IS SELECT EMPNO,SAL FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP UPDATE EMP SET SAL=SAL+500 WHERE CURRENT OF c_emp; END LOOP; END;

Q18: Increase salary by 10% using WHERE CURRENT OF

DECLARE CURSOR c_emp IS SELECT EMPNO,SAL FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP UPDATE EMP SET SAL=SAL*1.1 WHERE CURRENT OF c_emp; END LOOP; END;

Q19: Delete employees with salary < 2000 using cursor (Updatable Cursor + DELETE)

DECLARE CURSOR c_emp IS SELECT EMPNO,SAL FROM EMP WHERE SAL<2000 FOR UPDATE; BEGIN FOR rec IN c_emp LOOP DELETE FROM EMP WHERE CURRENT OF c_emp; END LOOP; END;

Q20: Update job of employees using updatable cursor (Updatable Cursor)

DECLARE CURSOR c_emp IS SELECT EMPNO,JOB FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP UPDATE EMP SET JOB='ANALYST' WHERE CURRENT OF c_emp; END LOOP; END;

Q21: Fetch records using cursor and modify values row-by-row (Updatable Cursor)

DECLARE CURSOR c_emp IS SELECT EMPNO,SAL FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP UPDATE EMP SET SAL=SAL+100 WHERE CURRENT OF c_emp; DBMS_OUTPUT.PUT_LINE('Updated Salary for Empno '||rec.EMPNO); END LOOP; END;

Q22: Explicit cursor with IF condition to display high salary employees (Cursor + IF)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; BEGIN FOR rec IN c_emp LOOP IF rec.SAL>4000 THEN DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); END IF; END LOOP; END;

Q23: Cursor FOR LOOP to calculate bonus (10% of salary)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; BEGIN FOR rec IN c_emp LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME||' Bonus='||(rec.SAL*0.1)); END LOOP; END;

Q24: Parameterized cursor looping through multiple departments

DECLARE CURSOR c_emp(dno NUMBER) IS SELECT ENAME FROM EMP WHERE DEPTNO=dno; BEGIN FOR rec IN c_emp(10) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME); END LOOP; FOR rec IN c_emp(20) LOOP DBMS_OUTPUT.PUT_LINE(rec.ENAME); END LOOP; END;

Q25: Update and display employee data using updatable cursor (Updatable Cursor + Output)

DECLARE CURSOR c_emp IS SELECT EMPNO,ENAME,SAL FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP UPDATE EMP SET SAL=SAL+200 WHERE CURRENT OF c_emp; DBMS_OUTPUT.PUT_LINE('Updated '||rec.ENAME); END LOOP; END;

Q26: Display top 3 highest paid employees using cursor (Cursor + Logic)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP ORDER BY SAL DESC; i NUMBER:=0; BEGIN FOR rec IN c_emp LOOP EXIT WHEN i=3; DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); i:=i+1; END LOOP; END;

Q27: Simulate pagination (fetch limited rows using cursor)

DECLARE CURSOR c_emp IS SELECT ENAME,SAL FROM EMP; i NUMBER:=0; BEGIN FOR rec IN c_emp LOOP EXIT WHEN i=5; -- fetch first 5 rows DBMS_OUTPUT.PUT_LINE(rec.ENAME||' '||rec.SAL); i:=i+1; END LOOP; END;

Q28: Copy employee data into another table using cursor (Cursor + INSERT)

DECLARE CURSOR c_emp IS SELECT EMPNO,ENAME,SAL FROM EMP; BEGIN FOR rec IN c_emp LOOP INSERT INTO EMP_COPY VALUES(rec.EMPNO,rec.ENAME,rec.SAL); END LOOP; END;

Q29: Display department-wise employee count using cursor (Cursor + Grouping Logic)

DECLARE CURSOR c_emp IS SELECT DEPTNO,COUNT(*) AS cnt FROM EMP GROUP BY DEPTNO; BEGIN FOR rec IN c_emp LOOP DBMS_OUTPUT.PUT_LINE('Dept '||rec.DEPTNO||' Count='||rec.cnt); END LOOP; END;

Q30: Process employee records and apply salary increments based on job (Cursor + Business Logic)

DECLARE CURSOR c_emp IS SELECT EMPNO,ENAME,JOB,SAL FROM EMP FOR UPDATE; BEGIN FOR rec IN c_emp LOOP IF rec.JOB='CLERK' THEN UPDATE EMP SET SAL=SAL+200 WHERE CURRENT OF c_emp; ELSIF rec.JOB='MANAGER' THEN UPDATE EMP SET SAL=SAL+500 WHERE CURRENT OF c_emp; ELSE UPDATE EMP SET SAL=SAL+300 WHERE CURRENT OF c_emp; END IF; DBMS_OUTPUT.PUT_LINE('Updated '||rec.ENAME); END LOOP; END;
